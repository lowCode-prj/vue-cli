# 引入echarts
    npm install echarts --save
# 安装指定低版本的less-loader（高版本的会出现错误）
    npm install less-loader@4.1.0 --save
# npm安装less 
    npm install less 
# 在vue.config.js配置（ 跳过eslint每次保存都启用验证）
    lintOnSave:false
#  引入组件
    复制组件到components下并在APP组件下引入、注册，使用
#  终端执行 
    安装依赖：npm i 
    执行：npm run serve
